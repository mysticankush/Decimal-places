#include<stdio.h>

int main() {
   int n,p;
 	scanf("%d%d",&n,&p);
 	int l=0,r=n;
 	double ans;
 	while(l<=r)
 	{
 		int mid=(l+r)/2;
 		if(mid*mid==n)
 		{
 			ans=mid;
 			break;
 		}
 		if(mid*mid>n)
 		{
 			r=mid-1;
 		}
 		else
 		{
 			ans=mid;
 			l=mid+1;
 		}
 	}
 	double temp=0.1;
 	while(p--)
 	{
 		double ans1=ans;
 		for(int i=0;i<=9;i++)
 		{
 			double tempans=temp*i;
 			double num=tempans+ans1;
 			if(num*num<=n)
 			{
 				ans=num;
 			}
 			else
 			break;
 		}
 		temp/=10;
 		
 	}
 	printf("%0.10g",ans);
}
